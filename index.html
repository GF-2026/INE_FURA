document.getElementById('btnPDF').addEventListener('click', () => { if (validarFormulario()) { // Obtiene los valores de nombre y apellidos const nombre = document.querySelector('.nameComplete input')?.value.trim() || ''; const apellidoP = document.querySelector('.AddressComplete input')?.value.trim() || ''; const apellidoM = document.querySelector('.lastname2-container1 input')?.value.trim() || ''; // Crea un nombre de archivo limpio (sin espacios ni caracteres especiales) const nombreArchivo = `FURA_${[nombre, apellidoP, apellidoM] .filter(Boolean) .join('_') .replace(/\s+/g, '_') .toUpperCase()}.pdf`; generatePDF(nombreArchivo); } });
